# shadowsocks搭建

# 下载执行脚本

## 下载wget
yum install wget

## 下载shell脚本
wget –no-check-certificate -O shadowsocks.sh https://raw.githubusercontent.com/teddysun/shadowsocks_install/master/shadowsocks.sh

# 修改权限

chmod +x shadowsocks.sh

# 执行脚本

./shadowsocks.sh 2>&1 | tee shadowsocks.log

以下为我安装过程中遇到的错误及解决方案。

# gcc环境

yum install gcc

# libsodium环境

报错libsodium install failed!,缺少libsodium环境

 yum install epel-release -y
 yum install libsodium -y

# make环境

 yum install make -y

# python环境

shadowsocks.sh基于python编写，需要安装python环境

  • 如果系统安装的python3，需要将shell脚本中python命令改成python3

  • 同时将dist-packages文件所在目录，拷贝到/usr/local/lib/目录下

    cp -r /usr/lib/python3.6 /usr/local/lib/.

# 客户端

shadowsocks-windows

完成配置后，启用系统代理，选择PAC模式。

# 使用BBR加速上网

##下载bbr
wget --no-check-certificate https://github.com/teddysun/across/raw/master/bbr.sh

##修改权限
chmod +x bbr.sh

##启动bbr
./bbr.sh

##检查是否启动
lsmod | grep bbr
